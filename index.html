CODEWARS Challenge "Variations in an Array" .

Challenge Write a function that takes an array of numbers and returns a new array containing all possible combinations obtained by increasing each number by 1, and also leaving it unchanged.

function variations(arr) {
            const results = [];
            
            // Helper function to generate variations
            function generate(current, index) {
                if (index === arr.length) {
                    results.push(current);
                    return;
                }
                
                // Take the current number as is
                generate([...current, arr[index]], index + 1);
                
                // Take the current number + 1
                generate([...current, arr[index] + 1], index + 1);
            }
            
            generate([], 0);
            return results;
        }
        const input = [1, 2];
        const output = variations(input);
        console.log(output);
